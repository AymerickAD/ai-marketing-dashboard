# Modèle de prédiction d'engagement social media
import pandas as pd
from sklearn.ensemble import RandomForestRegressor
from sklearn.model_selection import train_test_split
import numpy as np

class SocialMediaPredictor:
    def __init__(self):
        self.model = RandomForestRegressor(n_estimators=100, random_state=42)
        self.features = ['hour_posted', 'day_of_week', 'content_length', 
                        'hashtags_count', 'mentions_count', 'sentiment_score']
    
    def preprocess_data(self, df):
        """Préparation des données pour le modèle"""
        df['hour_posted'] = pd.to_datetime(df['timestamp']).dt.hour
        df['day_of_week'] = pd.to_datetime(df['timestamp']).dt.dayofweek
        df['content_length'] = df['content'].str.len()
        df['hashtags_count'] = df['content'].str.count('#')
        df['mentions_count'] = df['content'].str.count('@')
        return df[self.features]
    
    def train(self, data):
        """Entraînement du modèle"""
        X = self.preprocess_data(data)
        y = data['engagement_rate']
        X_train, X_test, y_train, y_test = train_test_split(X, y, test_size=0.2)
        
        self.model.fit(X_train, y_train)
        accuracy = self.model.score(X_test, y_test)
        return accuracy
    
    def predict_engagement(self, content_data):
        """Prédiction du taux d'engagement"""
        X = self.preprocess_data(content_data)
        prediction = self.model.predict(X)
        return prediction[0]
